1. Add a global scroll → CSS variable hook

Create a tiny hook, e.g. useScrollLight.ts:

import { useEffect } from "react";

export function useScrollLight() {
  useEffect(() => {
    let ticking = false;

    const onScroll = () => {
      if (!ticking) {
        window.requestAnimationFrame(() => {
          const max = document.body.scrollHeight - window.innerHeight || 1;
          const ratio = window.scrollY / max; // 0 → 1
          // Expose a normalized scroll value as a CSS variable
          document.documentElement.style.setProperty(
            "--scroll-ratio",
            ratio.toString()
          );
          ticking = false;
        });
        ticking = true;
      }
    };

    window.addEventListener("scroll", onScroll, { passive: true });
    onScroll(); // initialize

    return () => window.removeEventListener("scroll", onScroll);
  }, []);
}


Then, in your FeedScreen component, just call it once:

import React from "react";
import { useScrollLight } from "../hooks/useScrollLight";

export const FeedScreen: React.FC = () => {
  useScrollLight();

  return (
    <div className="feed-root">
      {/* your existing feed layout */}
    </div>
  );
};

2. Global background that “tilts” with scroll

In your main CSS (or Tailwind layer), define a liquid background that shifts based on --scroll-ratio:

/* Root background: red/white/blue liquid gradient */
.feed-root {
  min-height: 100vh;
  position: relative;
  padding-bottom: 80px; /* for bottom nav */
  background:
    radial-gradient(circle at calc(20% + 40% * var(--scroll-ratio, 0))
                            calc(20% + 30% * var(--scroll-ratio, 0)),
                    rgba(255, 255, 255, 0.18),
                    transparent 55%),
    radial-gradient(circle at calc(80% - 30% * var(--scroll-ratio, 0))
                            calc(60% - 20% * var(--scroll-ratio, 0)),
                    rgba(230, 57, 58, 0.45),
                    transparent 60%),
    radial-gradient(circle at 50% 100%,
                    rgba(59, 91, 169, 0.65),
                    #050b1b 70%);
  background-attachment: fixed;
  background-size: cover;
}


This gives you a “moving” light blob that subtly shifts as you scroll, and because it’s just a couple gradients + one CSS variable, it’s cheap.

3. Liquid glass cards with scroll-reactive reflections

Now give each card a pseudo-element that picks up that same motion, so it looks like the light bends around the rounded corners.

.glass-card {
  position: relative;
  border-radius: 22px;
  background: rgba(7, 16, 40, 0.45);
  backdrop-filter: blur(18px);
  -webkit-backdrop-filter: blur(18px);
  border: 1px solid rgba(255, 255, 255, 0.16);
  box-shadow: 0 18px 30px rgba(0, 0, 0, 0.35);
  overflow: hidden;
}

/* “Reflection” overlay */
.glass-card::before {
  content: "";
  position: absolute;
  inset: -20%;
  pointer-events: none;
  mix-blend-mode: screen;
  opacity: 0.35;
  background:
    radial-gradient(circle at calc(30% + 40% * var(--scroll-ratio, 0))
                            calc(0% + 40% * var(--scroll-ratio, 0)),
                    rgba(255, 255, 255, 0.75),
                    transparent 55%);
  transition: opacity 150ms ease-out;
}

/* Slightly stronger reflection on touch/hover */
.glass-card:active::before,
.glass-card:hover::before {
  opacity: 0.5;
}


Then make sure all main feed tiles use that class:

<div className="grid grid-cols-2 gap-4">
  <article className="glass-card">{/* SHORT */}</article>
  <article className="glass-card">{/* NEWS */}</article>
  <article className="glass-card">{/* POLL */}</article>
  <article className="glass-card">{/* PETITION */}</article>
</div>
